Inductive typed_target_term : Set :=
  typed_target_term_cons : ∀ (t : Target.trm) (T : Target.typ) (Htyp : Target.ty_trm empty t T), typed_target_term.

  match R with
  | typed_target_term_cons t _ _ ⇒ t
  end.
