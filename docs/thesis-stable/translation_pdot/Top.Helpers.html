Lemma intersection_order : ∀ G A1 B1 A2 B2,
    G ⊢ A1 <: A2 →
    G ⊢ B1 <: B2 →
    G ⊢ A1 ∧ B1 <: A2 ∧ B2.
Proof.
  intros.
  apply subtyp_and2.
  - apply subtyp_trans with A1; auto.
  - apply subtyp_trans with B1; auto.
Qed.

Lemma swap_typ : ∀ G X Y L T,
    G ⊢ trm_path (pvar Y) : {{pvar X}} →
    G ⊢ trm_path (pvar X) : T →
    G ⊢ pvar X ↓ L =:= pvar Y ↓ L.
Proof.
  intros.
  constructor.
  - eapply subtyp_sngl_qp with (p := pvar Y) (q := pvar X); eauto.
    assert (HR: pvar X = (pvar X) •• []) by crush.
    rewrite HR at 2.
    assert (HR2: pvar Y = (pvar Y) •• []) by crush.
    rewrite HR2 at 2.
    apply rpath.
  - eapply subtyp_sngl_pq with (p := pvar Y) (q := pvar X); eauto.
    assert (HR: pvar X = (pvar X) •• []) by crush.
    rewrite HR at 2.
    assert (HR2: pvar Y = (pvar Y) •• []) by crush.
    rewrite HR2 at 2.
    apply rpath.
Qed.

Lemma sub_and_assoc : ∀ G A B C,
    G ⊢ A ∧ (B ∧ C) <: (A ∧ B) ∧ C.
Proof.
  intros.
  apply subtyp_and2.
  + apply subtyp_and2.
    × apply subtyp_and11.
    × eapply subtyp_trans.
      -- apply subtyp_and12.
      -- apply subtyp_and11.
  + eapply subtyp_trans;
      apply subtyp_and12.
Qed.
