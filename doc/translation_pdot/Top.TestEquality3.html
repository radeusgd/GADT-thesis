Lemma construct_tuple_lemma : ∀ G A B C D,
    G ⊢ A =:= C →
    G ⊢ B =:= D →
    G ⊢ (pvar lib) ↓ Tuple ∧ {T1 == A} ∧ {T2 == B} =:= (pvar lib) ↓ Tuple ∧ {T1 == C} ∧ {T2 == D}.
Proof.
  introv [] [].
  repeat apply¬ eq_and_merge.
Qed.
